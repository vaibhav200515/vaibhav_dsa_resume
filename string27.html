#include < stdio.h>
#include < string.h>
#define size 20

void textalign(char[]);

int main(){
    int value;
    char str[size];
    printf("Enter a sentence: ");
    fgets(str, size, stdin);
    str[strcspn(str, "\n")] = '\0';
    textalign(str);
    return 0;
}

void textalign(char str[]){
    int value;
    printf("Enter 1 to align text to left \n2 to align text to center \n3 to align text to right: ");
    scanf("%d", &value);
    switch (value){
        case 1:
            puts(str);   
            break;
        case 2:
            printf("\t\t\t\t\t");
            puts(str);
            break;
        case 3:
            printf("\t\t\t\t\t\t\t\t\t\t");
            puts(str);
            break;
        default:
            printf("Invalid Input!");
    }
}
