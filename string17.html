#include < stdio.h>
#include < string.h>

#define size 10

int vowel(char[]);

int main(){
    char str[size], count;
    printf("Enter input in string: ");
    fgets(str, size, stdin);
    str[strcspn(str, "\n")] = '\0';
    count=vowel(str);
    printf("Number of vowel: %d", count);
    return 0;
}

int vowel(char str[]){
    int count=0, i=0;
    while(str[i]!='\0'){
        if(str[i]=='a' || str[i]=='A' || str[i]=='e' || str[i]=='E' || str[i]=='i' || str[i]=='I' || str[i]=='o' || str[i]=='O' || str[i]=='u' || str[i]=='U'){
            count++;
        }
        i++;
    }
    return count;
}

Enter input in string: atul
Number of vowel: 2
