#include < stdio.h>
#include < string.h>

#define size 20

int search(char[], char[]);

int main(){
    int result;
    char str[size], value[size];
    printf("Enter input in string: ");
    fgets(str, size, stdin);
    str[strcspn(str, "\n")] = '\0';
    printf("Enter value to be searched: ");
    fgets(value, size, stdin);
    value[strcspn(value, "\n")] = '\0';
    result=search(str, value);
    if(result==0){
        printf("Word not found! ");
    }
    else{
        printf("Entered word occurs %d time.", result);
    }
    return 0;
}

int search(char str[], char value[]){
    int count=0, i=0, j=0;
    while(str[i]!='\0'){
        if(str[i]==' '){
            i++;
            continue;
        }
        else if(value[j]==str[i]){
            while(value[j]!='\0' || str[i]!='\0'){
                if(value[j]==str[i]){
                    j++;
                    i++;
                }
                else{
                    break;
                }
            }
        }
        if(value[j]=='\0'){
            count++;
        }
        i++;
        j=0;
    }
    return count;
}
