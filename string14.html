int length(char str[]){
    int i=0;
    while(str[i]!='\0'){
        i++;
    }
    return i;
}

void difference(char str[], int len){
    int i=0, diff;
    for(i=0; i < len-1; i++){
        diff=str[i]-str[i+1];
        printf("Differnce of %d and %d character is: %d\n", i+1, i+2, diff);
    }
}
