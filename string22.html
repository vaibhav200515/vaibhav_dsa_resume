    #include < stdio.h>
    #include < string.h>
    
    #define size 10
    
    void bubsort(char[], int);
    int length(char[]);
    
    int main(){
        int len;
        char str[size];
        printf("Enter input in string: ");
        fgets(str, size, stdin);
        str[strcspn(str, "\n")] = '\0';
        len=length(str);
        bubsort(str, len);
        puts(str);
        return 0;
    }
    
    int length(char str[]){
        int i=0;
        while(str[i]!='\0'){
            i++;
        }
        return i;
    }
    void bubsort(char str[], int len)
    {
        int i, j;
        char temp;
        for(i=0; i < len-1; i++){
            for(j=0; j < len-1-i; j++){
                if(str[j]> str[j+1]){
                   temp=str[j];
                   str[j]=str[j+1];
                   str[j+1]=temp; 
                }
            }
        }
    }
