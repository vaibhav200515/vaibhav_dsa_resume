#include  < stdio.h>

int main(){
    int arr[4][2], i, j;
    printf("Enter value into matrix: ");
    for(i=0; i < 4; i++){
        for(j=0; j < 2; j++){
            scanf("%d", &arr[i][j]);
        }
    }
    printf("Entered matrix is: \n");
    for(i=0; i < 4; i++){
        for(j=0; j < 2; j++){
            printf(" %d ", arr[i][j]);
        }
        printf("\n");
    }
    return 0;
}
