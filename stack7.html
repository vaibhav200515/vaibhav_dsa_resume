                #include < stdio.h>
            #include < stdlib.h>
            
            #define MAX_SIZE 100
            
            int stack[MAX_SIZE];
            int top1 = -1;
            int top2 = MAX_SIZE;
            
            int isFull() {
                return top1 == top2 - 1;
            }
            
            int isEmpty(int stackNumber) {
                if (stackNumber == 1) {
                    return top1 == -1;
                } else if (stackNumber == 2) {
                    return top2 == MAX_SIZE;
                } else {
                    return 1; // Invalid stack number
                }
            }
            
            void push(int stackNumber, int item) {
                if (isFull()) {
                    printf("Stack %d is full. Cannot push %d.\n", stackNumber, item);
                } else if (stackNumber == 1) {
                    stack[++top1] = item;
                    printf("Pushed %d onto Stack %d.\n", item, stackNumber);
                } else if (stackNumber == 2) {
                    stack[--top2] = item;
                    printf("Pushed %d onto Stack %d.\n", item, stackNumber);
                } else {
                    printf("Invalid stack number.\n");
                }
            }
            
            void pop(int stackNumber) {
                if (isEmpty(stackNumber)) {
                    printf("Stack %d is empty. Cannot pop.\n", stackNumber);
                } else if (stackNumber == 1) {
                    printf("Popped %d from Stack %d.\n", stack[top1], stackNumber);
                    top1--;
                } else if (stackNumber == 2) {
                    printf("Popped %d from Stack %d.\n", stack[top2], stackNumber);
                    top2++;
                } else {
                    printf("Invalid stack number.\n");
                }
            }
            
            int peek(int stackNumber) {
                if (isEmpty(stackNumber)) {
                    printf("Stack %d is empty. No element to peek.\n", stackNumber);
                    return -1;
                } else if (stackNumber == 1) {
                    printf("Top element of Stack %d: %d\n", stackNumber, stack[top1]);
                    return stack[top1];
                } else if (stackNumber == 2) {
                    printf("Top element of Stack %d: %d\n", stackNumber, stack[top2]);
                    return stack[top2];
                } else {
                    printf("Invalid stack number.\n");
                    return -1;
                }
            }
            
            int main() {
                int choice, stackNumber, item;
            
                while (1) {
                    printf("Stack Operations:\n");
                    printf("1. Push\n");
                    printf("2. Pop\n");
                    printf("3. Peek\n");
                    printf("4. Is Empty\n");
                    printf("5. Exit\n");
            
                    printf("Enter your choice: ");
                    scanf("%d", &choice);
            
                    switch (choice) {
                        case 1:
                            printf("Enter stack number (1 or 2): ");
                            scanf("%d", &stackNumber);
                            printf("Enter an item to push: ");
                            scanf("%d", &item);
                            push(stackNumber, item);
                            break;
            
                        case 2:
                            printf("Enter stack number (1 or 2): ");
                            scanf("%d", &stackNumber);
                            pop(stackNumber);
                            break;
            
                        case 3:
                            printf("Enter stack number (1 or 2): ");
                            scanf("%d", &stackNumber);
                            peek(stackNumber);
                            break;
            
                        case 4:
                            printf("Enter stack number (1 or 2): ");
                            scanf("%d", &stackNumber);
                            if (isEmpty(stackNumber)) {
                                printf("Stack %d is empty.\n", stackNumber);
                            } else {
                                printf("Stack %d is not empty.\n", stackNumber);
                            }
                            break;
            
                        case 5:
                            exit(0);
                            break;
            
                        default:
                            printf("Invalid choice. Please enter a valid option.\n");
                    }
                }
            
                return 0;
            }
