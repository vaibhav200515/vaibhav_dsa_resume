#include < stdio.h>

#define size 50

void search(char[]);

int main(){
    char str[size];
    printf("Enter input and press ';' to end taking input: ");
    scanf("%[^;]", str);
    search(str);
    return 0;
}

void search(char str[]){
    int space=0, character=0, words=0, lines=1, i=0;
    while(str[i]!='\0'){
        if(str[i]==' '){
            space++;
            words++;
        }
        else if(str[i]=='\n'){
            lines++;
            words++;
        }
        character++;
        i++;
    }
    printf("Number of charcters: %d\n", character);
    printf("Number of space: %d\n", space);
    printf("Number of words: %d\n", words);
    printf("Number of lines: %d\n", lines);
}
