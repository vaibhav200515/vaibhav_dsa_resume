        #include < stdio.h>

        int fibonacci(int n);
        
        int main() {
            int num;
        
            printf("Enter a number to calculate its Fibonacci: ");
            scanf("%d", &num);
        
            if (num < 0) {
                printf("Fibonacci is not defined for negative numbers.\n");
            } else {
                printf("Fibonacci of %d is: %d\n", num, fibonacci(num));
            }
        
            return 0;
        }
        
        int fibonacci(int n) {
            if (n == 0) {
                return 0;
            } else if (n == 1) {
                return 1;
            } else {
                return fibonacci(n - 1) + fibonacci(n - 2);
            }
        }
