    #include stdio.h>
    #include stdlib.h>
    
    #define MAX_SIZE 100
    
    typedef struct {
        int items[MAX_SIZE];
        int top;
    } Stack;
    
    void initialize(Stack* stack) {
        stack->top = -1;
    }
    
    int isEmpty(Stack* stack) {
        return stack->top == -1;
    }
    
    int isFull(Stack* stack) {
        return stack->top == MAX_SIZE - 1;
    }
    
    void push(Stack* stack, int data) {
        if (!isFull(stack)) {
            stack->items[++stack->top] = data;
        }
    }
    
    int pop(Stack* stack) {
        if (!isEmpty(stack)) {
            return stack->items[stack->top--];
        }
        return -1;
    }
    
    void sortStack(Stack* stack) {
        int tempStack[MAX_SIZE];
        int tempTop = -1;
    
        while (!isEmpty(stack)) {
            int temp = pop(stack);
            while (!isEmpty(stack) && temp < stack->items[stack->top]) {
                push(&tempStack, pop(stack));
            }
            push(stack, temp);
            while (tempTop >= 0) {
                push(stack, tempStack[tempTop--]);
            }
        }
    }
    
    int main() {
        Stack stack;
        initialize(&stack);
    
        int n, data;
        
        printf("Enter the number of elements: ");
        scanf("%d", &n);
        
        printf("Enter %d elements:\n", n);
        for (int i = 0; i < n; i++) {
            scanf("%d", &data);
            push(&stack, data);
        }
    
        printf("Original stack: ");
        while (!isEmpty(&stack)) {
            printf("%d ", pop(&stack));
        }
        printf("\n");
    
        initialize(&stack);
    
        printf("Enter %d elements:\n", n);
        for (int i = 0; i < n; i++) {
            scanf("%d", &data);
            push(&stack, data);
        }
    
        sortStack(&stack);
    
        printf("Sorted stack: ");
        while (!isEmpty(&stack)) {
            printf("%d ", pop(&stack));
        }
        printf("\n");
    
        return 0;
    }
