        #include < stdio.h>
        #include < stdlib.h>
        #include < time.h>
        
        int binarySearch(int arr[], int low, int high, int key) {
            while (low <= high) {
                int mid = low + (high - low) / 2;
        
                if (arr[mid] == key)
                    return mid;
        
                if (arr[mid] < key)
                    low = mid + 1;
                else
                    high = mid - 1;
            }
        
            return -1;
        }
        
        double measureBinarySearchTime(int arr[], int size, int key) {
            clock_t start, end;
            double cpu_time_used;
        
            start = clock();
            binarySearch(arr, 0, size - 1, key);
            end = clock();
        
            cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
        
            return cpu_time_used;
        }
        
        int main() {
            int size, key;
        
            printf("Enter the size of the array: ");
            scanf("%d", &size);
        
            int *arr = (int *)malloc(size * sizeof(int));
        
            if (!arr) {
                printf("Memory allocation failed.\n");
                return 1;
            }
        
            printf("Enter %d elements in ascending order:\n", size);
        
            for (int i = 0; i < size; ++i)
                scanf("%d", &arr[i]);
        
            printf("Enter the key to search: ");
            scanf("%d", &key);
        
            int worstCaseIndex = binarySearch(arr, 0, size - 1, key);
            int bestCaseIndex = (worstCaseIndex != -1) ? worstCaseIndex : 0;
        
            printf("Worst-case: ");
            if (worstCaseIndex != -1)
                printf("Element %d found at index %d.\n", key, worstCaseIndex);
            else
                printf("Element %d not found.\n", key);
        
            printf("Best-case: ");
            if (bestCaseIndex != -1)
                printf("Element %d found at index %d.\n", key, bestCaseIndex);
            else
                printf("Element %d not found.\n", key);
        
            double worstCaseTime = measureBinarySearchTime(arr, size, key);
            double bestCaseTime = (worstCaseIndex != -1) ? worstCaseTime : 0.0;
        
            printf("Worst-case: Time taken for binary search: %f seconds.\n", worstCaseTime);
            printf("Best-case: Time taken for binary search: %f seconds.\n", bestCaseTime);
        
            free(arr);
        
            return 0;
        }
