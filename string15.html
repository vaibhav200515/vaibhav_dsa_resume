#include < stdio.h>
#include < string.h>

#define size 10

void search(char[], char);

int main(){
    char str[size], value;
    printf("Enter input in string: ");
    fgets(str, size, stdin);
    str[strcspn(str, "\n")] = '\0';
    printf("Enter value to be searched: ");
    scanf("%c", &value);
    search(str, value);
    return 0;
}

void search(char str[], char value){
    int result=0, i=0;
    while(str[i]!='\0'){
        if(value==str[i]){
            result=1;
            break;
        }
        else{
            result=0;
        }
        i++;
    }
    if(result==0){
        printf("Character not found! ");
    }
    else{
        printf("Character found! ");
    }
}
