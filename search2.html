        #include < stdio.h>

        int linearSearch(char str[], char key, int index) {
            if (str[index] == '\0') {
                return -1; // Key not found
            }
            if (str[index] == key) {
                return index; // Key found at current index
            }
            return linearSearch(str, key, index + 1); // Continue searching in the rest of the string
        }
        
        int main() {
            char str[50], key;
            int result;
        
            printf("Enter a string: ");
            scanf("%s", str);
        
            printf("Enter a character to search: ");
            scanf(" %c", &key);
        
            result = linearSearch(str, key, 0);
        
            if (result != -1) {
                printf("Character '%c' found at index %d.\n", key, result);
            } else {
                printf("Character '%c' not found in the string.\n", key);
            }
        
            return 0;
        }
